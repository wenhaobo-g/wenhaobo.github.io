    <h1>Tile Match Game</h1>
    <div class="board" id="board"></div>

    <script>
        const colors = ['red','red','blue','blue','green','green','yellow','yellow'];
        colors.sort(()=>Math.random()-0.5);
        const b = document.getElementById('board');
        let first = null;

        colors.forEach(c => {
            const d = document.createElement('div');
            d.className = 'card';
            d.onclick = () => flip(d, c);
            b.appendChild(d);
        });

        function flip(el, c) {
            el.style.background = c;
            if (!first) first = { el, c };
            else {
                if (first.c === c) {
                    first.el.onclick = null;
                    el.onclick = null;
                } else {
                    setTimeout(()=>{
                        first.el.style.background='#4fc3f7';
                        el.style.background='#4fc3f7';
                    },500);
                }
                first = null;
            }
        }
    </script>
