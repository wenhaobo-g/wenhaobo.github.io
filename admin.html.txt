        .game { background:#1e1e1e; padding:1rem; margin:8px 0; border-radius:6px; display:flex; justify-content:space-between; }

<body>
    <div class="container">
        <h2>Admin - Add / Delete Games</h2>
        <input id="gName" placeholder="Game Name">
        <input id="gDesc" placeholder="Description">
        <input id="gLink" placeholder="Page URL e.g. game-new.html">
        <button onclick="addGame()">Add Game</button>

        <div id="adminList"></div>
    </div>

    <script>
        let games = JSON.parse(localStorage.getItem('games')) || [];

        function renderAdmin() {
            const list = document.getElementById('adminList');
            list.innerHTML = '';
            games.forEach((g, i) => {
                list.innerHTML += `
                <div class="game">
                    <div>${g.name} - ${g.desc}</div>
                    <button onclick="del(${i})">Delete</button>
                </div>`;
            });
        }

        function addGame() {
            const name = document.getElementById('gName').value;
            const desc = document.getElementById('gDesc').value;
            const link = document.getElementById('gLink').value;
            games.push({ id:Date.now(), name, desc, link });
            localStorage.setItem('games', JSON.stringify(games));
            renderAdmin();
        }

        function del(index) {
            games.splice(index, 1);
            localStorage.setItem('games', JSON.stringify(games));
            renderAdmin();
        }

        renderAdmin();
    </script>
